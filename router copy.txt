import { runInit } from './commands/init.js';
import { runDev } from './commands/dev.js';
import { runDeploy } from './commands/deploy.js';
import { runDoctor } from './commands/doctor.js';
import { runGit } from './commands/git.js';

type CommandHandler = (params: string[]) => Promise<void>;

const COMMANDS: Record<string, CommandHandler> = {
  init: runInit,
  dev: runDev,
  deploy: runDeploy,
  doctor: runDoctor,
  git: runGit,
};

export async function route(command: string, params: string[]): Promise<void> {
  const handler = COMMANDS[command];
  if (!handler) {
    throw new Error(`Unknown command: "${command}". Run "dez --help" for usage.`);
  }
  await handler(params);
}
